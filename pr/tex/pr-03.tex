\section{Фундамент (1--4 балла)}

\textbf{Задача 1 (1 балл).} Упростите: $\sin^2 \alpha + \cos^2 \alpha$

\textbf{Задача 2 (2 балла).} Упростите: $\tg \alpha \cdot \ctg \alpha$

\textbf{Задача 3 (3 балла).} Найдите $\cos \alpha$, если $\sin \alpha = \dfrac{3}{5}$ и $0^\circ < \alpha < 90^\circ$.

\textbf{Задача 4 (4 балла).} Найдите $\tg \alpha$, если $\sin \alpha = 0{,}6,\ \cos \alpha = 0{,}8$.

\section{Стены (5--7 баллов)}

\textbf{Задача 5 (5 баллов).} Упростите: $(\sin \alpha + \cos \alpha)^2 - 2\sin \alpha \cos \alpha$

\textbf{Задача 6 (6 баллов).} Найдите $\sin \alpha$, если $\cos \alpha = -\dfrac{5}{13}$ и $90^\circ < \alpha < 180^\circ$.

\textbf{Задача 7 (7 баллов).} Докажите тождество: $\tg \alpha + \ctg \alpha = \dfrac{1}{\sin \alpha \cos \alpha}$

\section{Кровля (8--10 баллов)}

\textbf{Задача 8 (8 баллов).} Упростите: $(1 - \sin \alpha)(1 + \sin \alpha)$

\textbf{Задача 9 (9 баллов).} Зная, что $\tg \alpha = 2$, найдите: $\dfrac{\sin \alpha + \cos \alpha}{\sin \alpha - \cos \alpha}$

\textbf{Задача 10 (10 баллов).} Уклон кровли задан углом $\alpha$, для которого $\sin \alpha = 0{,}6$. Найдите $\tg \alpha$ — отношение высоты подъёма к горизонтальной проекции ската.
