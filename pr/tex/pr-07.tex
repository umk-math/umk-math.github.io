\section{Фундамент (1--4 балла)}

\textbf{Задача 1 (1 балл).} Решите уравнение: $\sqrt{x} = 3$

\textbf{Задача 2 (2 балла).} Решите уравнение: $\sqrt{x + 1} = 2$

\textbf{Задача 3 (3 балла).} Решите уравнение: $\sqrt{2x} = 6$

\textbf{Задача 4 (4 балла).} Решите уравнение: $\sqrt{x - 5} = 0$

\section{Стены (5--7 баллов)}

\textbf{Задача 5 (5 баллов).} Решите уравнение: $\sqrt{3x + 4} = 5$

\textbf{Задача 6 (6 баллов).} Решите уравнение: $\sqrt{x^2} = 4$

\textbf{Задача 7 (7 баллов).} Решите уравнение: $\sqrt{x + 6} = x$

\section{Кровля (8--10 баллов)}

\textbf{Задача 8 (8 баллов).} Решите уравнение: $\sqrt{x + 5} - \sqrt{x} = 1$

\textbf{Задача 9 (9 баллов).} Решите уравнение: $\sqrt[3]{x + 7} = 2$

\textbf{Задача 10 (10 баллов).} Диагональ квадратного фундамента: $d = \sqrt{2S}$, где S — площадь. Найдите S, если $d = 10$ м.
