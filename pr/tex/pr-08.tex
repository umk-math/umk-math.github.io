\section{Фундамент (1--4 балла)}

\textbf{Задача 1 (1 балл).} Найдите производную: $f(x) = 5x^3$

\textbf{Задача 2 (2 балла).} Найдите производную: $f(x) = x^4 + 2x^3 - x$

\textbf{Задача 3 (3 балла).} Найдите производную: $f(x) = 7$

\textbf{Задача 4 (4 балла).} Найдите $f'(2)$, если $f(x) = x^2 + 3x$

\section{Стены (5--7 баллов)}

\textbf{Задача 5 (5 баллов).} Найдите производную: $f(x) = (x + 1)(x - 4)$

\textbf{Задача 6 (6 баллов).} Найдите производную: $f(x) = (2x + 1)(x^2 - 3)$

\textbf{Задача 7 (7 баллов).} Решите уравнение $f'(x) = 0$, если $f(x) = x^3 - 12x + 5$

\section{Кровля (8--10 баллов)}

\textbf{Задача 8 (8 баллов).} Найдите производную и вычислите $f'(-1)$: $f(x) = (x^2 + 2)(x - 5)$

\textbf{Задача 9 (9 баллов).} Объём бетонной смеси при твердении меняется по закону $V(t) = 2t^2 + 4t$ (t — часы). Найдите скорость изменения объёма через 3 часа.

\textbf{Задача 10 (10 баллов).} Прочность строительной конструкции зависит от толщины $x$ по формуле $P(x) = -x^2 + 12x$. При какой толщине прочность максимальна?

\newpage
\section*{Ответы}

\begin{enumerate}
  \item $f'(x) = 15x^2$
  \item $f'(x) = 4x^3 + 6x^2 - 1$
  \item $f'(x) = 0$
  \item $f'(2) = 7$
  \item $f'(x) = 2x - 3$
  \item $f'(x) = 6x^2 + 2x - 6$
  \item $x = -2;\ x = 2$
  \item $f'(x) = 3x^2 - 10x + 2;\ f'(-1) = 15$
  \item 16.
  \item толщина 6 единиц, $P_{\max} = 36$.
\end{enumerate}
