\section{Фундамент (1--4 балла)}

\textbf{Задача 1 (1 балл).} Найдите производную и критические точки: $f(x) = x^2 - 6x + 5$

\textbf{Задача 2 (2 балла).} Найдите промежутки возрастания и убывания: $f(x) = x^2 - 8x + 15$

\textbf{Задача 3 (3 балла).} Найдите производную: $f(x) = 3x^4 - 2x^2 + 7$

\textbf{Задача 4 (4 балла).} Найдите производную: $f(x) = (x + 2)(x - 3)$

\section{Стены (5--7 баллов)}

\textbf{Задача 5 (5 баллов).} Исследуйте функцию на экстремумы: $f(x) = x^3 - 3x^2 + 2$

\textbf{Задача 6 (6 баллов).} Найдите наибольшее и наименьшее значение: $f(x) = x^2 - 4x + 3$ на $[0;\ 5]$

\textbf{Задача 7 (7 баллов).} Исследуйте функцию: $f(x) = x^3 - 6x^2 + 9x + 1$

\section{Кровля (8--10 баллов)}

\textbf{Задача 8 (8 баллов).} Число 12 разбейте на два слагаемых так, чтобы их произведение было наибольшим.

\textbf{Задача 9 (9 баллов).} Найдите наибольшее значение функции: $f(x) = -x^2 + 6x - 5$

\textbf{Задача 10 (10 баллов).} Площадка у стены, 80 м забора. Найти максимальную площадь.

\newpage
\section*{Ответы}

\begin{enumerate}
  \item $f'(x) = 2x - 6$; критическая точка $x = 3$.
  \item убывает на $(-\infty;\ 4)$, возрастает на $(4;\ +\infty)$.
  \item $f'(x) = 12x^3 - 4x$.
  \item $f'(x) = 2x - 1$.
  \item максимум $(0;\ 2)$, минимум $(2;\ -2)$.
  \item $f_{\min} = -1,\ f_{\max} = 8$.
  \item максимум $(1;\ 5)$, минимум $(3;\ 1)$.
  \item 6 и 6, максимальное произведение = 36.
  \item $f_{\max} = 4$.
  \item 20 м × 40 м, $S_{\max} = 800$ м².
\end{enumerate}
