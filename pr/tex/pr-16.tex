    {Практическая работа №16}
    {Степень с рациональным показателем}
    {}

\section{Фундамент (1--4 балла)}

\textbf{Задача 1 (1 балл).} Представьте в виде корня: $5^{1/2}$

\textbf{Задача 2 (2 балла).} Представьте в виде степени: $\sqrt[3]{7}$

\textbf{Задача 3 (3 балла).} Вычислите: $8^{\frac{1}{3}}$

\textbf{Задача 4 (4 балла).} Вычислите: $16^{\frac{1}{2}}$

\section{Стены (5--7 баллов)}

\textbf{Задача 5 (5 баллов).} Упростите: $a^{\frac{1}{2}} \cdot a^{\frac{1}{3}}$

\textbf{Задача 6 (6 баллов).} Упростите: $\left(x^{\frac{1}{2}}\right)^4$

\textbf{Задача 7 (7 баллов).} Вычислите: $27^{\frac{2}{3}}$

\section{Кровля (8--10 баллов)}

\textbf{Задача 8 (8 баллов).} Упростите выражение: $\dfrac{a^{\frac{3}{4}} \cdot a^{\frac{1}{4}}}{a^{\frac{1}{2}}}$

\textbf{Задача 9 (9 баллов).} Вычислите: $16^{\frac{3}{4}}$

\textbf{Задача 10 (10 баллов).} Объём бетонного куба $V = a^3$. Выразите сторону $a$ через $V$ с помощью степени с рациональным показателем и найдите $a$, если $V = 125$ м³.

\newpage
\section*{Ответы}

\begin{enumerate}
  \item $\sqrt{5}$
  \item $7^{\frac{1}{3}}$
  \item 2.
  \item 4.
  \item $a^{\frac{5}{6}}$
  \item $x^2$
  \item 9.
  \item $a^{\frac{1}{2}}$ или $\sqrt{a}$
  \item 8.
  \item $a = V^{\frac{1}{3}};\ a = 5$ м.
\end{enumerate}
