    {Практическая работа №18}
    {Показательные неравенства}
    {}

\section{Фундамент (1--4 балла)}

\textbf{Задача 1 (1 балл).} Решите неравенство: $2^x > 8$

\textbf{Задача 2 (2 балла).} Решите неравенство: $3^{x-2} \le 27$

\textbf{Задача 3 (3 балла).} Решите неравенство: $\left(\frac{1}{2}\right)^x < 4$

\textbf{Задача 4 (4 балла).} Решите неравенство: $5^{2x} \ge 1$

\section{Стены (5--7 баллов)}

\textbf{Задача 5 (5 баллов).} Решите неравенство: $4^{x} - 6 \cdot 2^{x} + 8 \le 0$

\textbf{Задача 6 (6 баллов).} Решите неравенство: $9^{x} - 10 \cdot 3^{x} + 9 > 0$

\textbf{Задача 7 (7 баллов).} Решите неравенство: $2^{x+1} + 2^{2-x} \le 9$

\section{Кровля (8--10 баллов)}

\textbf{Задача 8 (8 баллов).} Количество кирпичей на стройке удваивается каждый день по закону $N(t) = 100 \cdot 2^{t}$, где $t$ — дни. Через сколько дней количество кирпичей превысит 3200 штук?

\textbf{Задача 9 (9 баллов).} Давление в трубе падает по закону $P(h) = 64 \cdot \left(\frac{1}{2}\right)^{\frac{h}{2}}$, где $h$ — высота в метрах. На какой высоте давление станет меньше 4 единиц?

\textbf{Задача 10 (10 баллов).} Решите неравенство: $\frac{3^{x} - 9}{2^{x} - 4} \ge 0$

\newpage
\section*{Ответы}

\begin{enumerate}
  \item $x \in (3; +\infty)$
  \item $x \in (-\infty; 5]$
  \item $x \in (-2; +\infty)$
  \item $x \in [0; +\infty)$
  \item $x \in [1; 2]$
  \item $x \in (-\infty; 0) \cup (2; +\infty)$
  \item $x \in [-1; 2]$
  \item через $6$ дней (на 6-й день количество превысит 3200)
  \item $h > 8$ метров
  \item $x \in (-\infty; 2) \cup (2; +\infty)$
\end{enumerate}
