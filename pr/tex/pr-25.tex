    {Практическая работа №25}
    {Логарифмические неравенства}
    {}

\section{Фундамент (1--4 балла)}

\textbf{Задача 1 (1 балл).} Решите неравенство: $\log_{2} x > 3$

\textbf{Задача 2 (2 балла).} Решите неравенство: $\log_{3} (x - 1) \le 2$

\textbf{Задача 3 (3 балла).} Решите неравенство: $\log_{\frac{1}{2}} x < 3$

\textbf{Задача 4 (4 балла).} Решите неравенство: $\log_{5} (2x + 1) \ge 0$

\section{Стены (5--7 баллов)}

\textbf{Задача 5 (5 баллов).} Решите неравенство: $\log_{2} (x + 3) + \log_{2} (x - 3) \ge 4$

\textbf{Задача 6 (6 баллов).} Решите неравенство: $\log_{3} (x^2 - 4) - \log_{3} (x + 2) < 1$

\textbf{Задача 7 (7 баллов).} Решите неравенство: $\log_{0{,}5}^{2} x - \log_{0{,}5} x - 2 \le 0$

\section{Кровля (8--10 баллов)}

\textbf{Задача 8 (8 баллов).} Решите неравенство: $\log_{2} (3x - 5) < \log_{2} (x + 3)$

\textbf{Задача 9 (9 баллов).} Решите неравенство: $\log_{3} (x^2 - 2x) \le 1$

\textbf{Задача 10 (10 баллов).} Решите неравенство: $\frac{\log_{2} x - 3}{\log_{2} x - 1} \ge 1$

\newpage
\section*{Ответы}

\begin{enumerate}
  \item $x \in (8; +\infty)$
  \item $x \in (1; 10]$
  \item $x \in \left(\frac{1}{8}; +\infty\right)$
  \item $x \in [0; +\infty)$
  \item $x \in [5; +\infty)$
  \item $x \in (2; 5)$
  \item $x \in [0{,}25; 2]$
  \item $x \in \left(\frac{5}{3}; 4\right)$
  \item $x \in [-1; 0) \cup (2; 3]$
  \item $x \in (0; 2)$
\end{enumerate}
